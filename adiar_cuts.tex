        \end{lemma}

        \begin{theorem}[\footnotesize Lampis, Kaouri, Mitsou 2011]

        \end{theorem}

    Given maximum $2$-level cuts size $C_f$ for $f$ and $C_g$ for $g$, the
    maximum $2$-level cut for $f \odot g$ is less than or equal to
    $C_f \cdot C_g$.
  \end{theorem}
